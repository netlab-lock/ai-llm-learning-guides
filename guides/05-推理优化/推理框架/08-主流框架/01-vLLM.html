# vllm/v1/core/sched/scheduler.py
class Scheduler:
    def schedule(self) -> SchedulerOutput:
        """每个 iteration 调用一次，决定哪些请求参与本轮计算"""

        scheduled_new: list[NewRequestData] = []      # 新请求
        scheduled_resumed: list[ResumedRequestData] = []  # 被恢复的请求
        scheduled_running: list[RunningRequestData] = []  # 继续运行的请求

        # ===== Phase 1: 处理 Waiting 队列中的新请求 =====
        while self.waiting:
            req = self.waiting[0]
            # 计算需要多少 KV blocks
            num_new_blocks = self._get_num_new_blocks(req)

            if num_new_blocks == 0:
                # 已有足够 blocks（前缀缓存命中）
                break

            # 尝试分配物理 blocks
            if self.block_manager.can_allocate(num_new_blocks):
                blocks = self.block_manager.allocate(num_new_blocks)
                req.block_table.extend(blocks)
                self.waiting.pop(0)
                scheduled_new.append(...)
            else:
                # 内存不足 → 尝试抢占
                if not self._preempt_running_req():
                    break  # 无法抢占，停止调度新请求

        # ===== Phase 2: Running 队列继续 Decode =====
        for req in self.running:
            # 每个 decode token 需要 1 个新 block（如果当前 block 已满）
            if req.is_block_full():
                new_block = self.block_manager.allocate(1)
                req.block_table.append(new_block)
            scheduled_running.append(...)

        # ===== Phase 3: 构造输出 =====
        return SchedulerOutput(
            scheduled_new_reqs=scheduled_new,
            scheduled_running_reqs=scheduled_running,
            # ... 其他字段
        )

5. 模型执行：ModelRunner.execute_model()

# vllm/v1/worker/gpu_model_runner.py
class GPUModelRunner:
    def execute_model(self, scheduler_output: SchedulerOutput):
        """执行一次前向传播，生成一个 token"""

        # 1) 准备输入 tensors
        input_ids = self._prepare_input_ids(scheduler_output)
        #    shape: [total_num_tokens] — 所有请求的 token 拼接

        positions = self._prepare_positions(scheduler_output)
        #    每个 token 在其序列中的位置

        # 2) ★ 构造 Block Table — PagedAttention 的核心
        block_tables = self._prepare_block_tables(scheduler_output)
        #    shape: [num_reqs, max_num_blocks_per_req]
    #    这个表告诉 attention kernel 去哪里读 KV

        # 3) 执行模型前向传播
        hidden_states = self.model(
            input_ids=input_ids,
            positions=positions,
            kv_caches=self.kv_caches,      # 物理 KV Cache blocks
            block_tables=block_tables,      # 逻辑→物理映射
        )

        # 4) 通过 LM Head 得到 logits → 采样下一个 token
        logits = self.model.compute_logits(hidden_states)
        output_token_ids = self.sampler.sample(logits, sampling_params)

        # 5) 更新序列状态
        for req, token_id in zip(running_reqs, output_token_ids):
            req.token_ids.append(token_id)
            if token_id == eos_token_id:
                req.set_finished()

        return ModelRunnerOutput(token_ids=output_token_ids)

2.4 PagedAttention CUDA Kernel 实现原理

2.4.1 CUDA Kernel 的内存访问模式

传统 Attention kernel 假设 K/V 在内存中是连续的，直接用偏移量访问。PagedAttention 修改了这个假设：

// 传统 Attention 的内存访问（简化）
// K_cache 地址 = base_ptr + seq_id * seq_stride + pos * head_dim
// → 直接线性访问，高效但要求连续内存

// PagedAttention 的内存访问（简化）
// 1. 查 Block Table 得到物理块号
int physical_block = block_table[seq_id][logical_block_id];
// 2. 计算物理地址
// K_cache 地址 = kv_cache_base + physical_block * block_size * head_dim
//              + offset_in_block * head_dim
// → 间接访问，但允许非连续存储

2.4.2 Attention Score 计算伪代码（带 Block 循环）

// PagedAttention 的核心 CUDA kernel（伪代码，简化自 vLLM 源码）
__global__ void paged_attention_v1_kernel(
    float* output,           // [num_seqs, num_heads, head_dim]
    const float* q,          // [num_seqs, num_heads, head_dim]
    const float* k_cache,    // 物理 KV Cache blocks
    const float* v_cache,
    const int* block_tables, // [num_seqs, max_num_blocks]
    const int* context_lens, // 每个序列的实际长度
    const int block_size,    // 每个 block 的 token 数 (e.g., 16)
    const int num_kv_heads,
    const int head_dim
) {
    // 每个 CUDA block 处理一个 (seq, head) 对
    int seq_id = blockIdx.x;
    int head_id = blockIdx.y;

    int context_len = context_lens[seq_id];
    int num_blocks = (context_len + block_size - 1) / block_size;

    // Query vector（本 seq 本 head 的 query）
    float* q_vec = q + (seq_id * num_heads + head_id) * head_dim;

    // 用于 softmax 的 online 算法变量
    float max_score = -INFINITY;
    float sum_exp = 0.0f;
    float acc[head_dim] = {0};  // 累积的加权 V

    // ★ 遍历所有逻辑块
    for (int block_idx = 0; block_idx < num_blocks; block_idx++) {
        // 通过 Block Table 查找物理块号
        int physical_block = block_tables[seq_id * max_blocks + block_idx];

        // 计算该 block 的有效 token 数（最后一块可能不满）
        int tokens_in_block = min(block_size, context_len - block_idx * block_size);

        // 遍历 block 内的每个 token
        for (int i = 0; i < tokens_in_block; i++) {
            // ★ 通过物理块号间接访问 K
            int k_offset = (physical_block * block_size + i) * head_dim;
            float* k_vec = k_cache + (head_id * total_blocks + k_offset);

            // 计算 attention score: q · k / sqrt(d_k)
            float score = dot_product(q_vec, k_vec, head_dim) / sqrtf(head_dim);

            // Online softmax（FlashAttention 风格）
            float new_max = max(max_score, score);
            float exp_old = expf(max_score - new_max);
            float exp_new = expf(score - new_max);
            sum_exp = sum_exp * exp_old + exp_new;
            max_score = new_max;

            // 累加加权 V
            // ★ 通过物理块号间接访问 V
            int v_offset = (physical_block * block_size + i) * head_dim;
            float* v_vec = v_cache + (head_id * total_blocks + v_offset);

            for (int d = 0; d < head_dim; d++) {
                acc[d] = acc[d] * exp_old + exp_new * v_vec[d];
            }
        }
    }

    // 归一化：acc / sum_exp
    for (int d = 0; d < head_dim; d++) {
        output[(seq_id * num_heads + head_id) * head_dim + d] = acc[d] / sum_exp;
    }
}

2.4.3 FlashAttention 与 PagedAttention 的结合

vLLM 实际使用的是 FlashAttention 的分块（tiling）算法 + PagedAttention 的间接寻址，两者的结合方式：

FlashAttention 的核心思想： PagedAttention 的核心思想： 将 Q/K/V 分成小块（tiles） K/V 不连续，通过 Block Table 间接访问 每个 tile 放在 SRAM 中计算 逻辑块 → 物理块的映射 避免写入巨大的 attention matrix 到 HBM ═══════════════ 结合方式 ═══════════════ 1. Q tile 放在 SRAM (寄存器/共享内存) 2. 遍历 K/V 的逻辑块: a. 通过 Block Table 查物理块号 b. 将该物理块的 K tile 加载到 SRAM c. 在 SRAM 中计算 Q·K^T (无 HBM 写入) d. Online softmax 更新 max/sum e. 加载 V tile 到 SRAM f. 累加 acc = acc * scale + softmax_score * V 3. 最终归一化，写回 HBM 关键：attention score 矩阵永远不出现在 HBM 中！ Block Table 的间接寻址增加了 ~5% 的地址计算开销， 但通过分块计算消除了 O(n²) 的 HBM 写入。
性能影响：PagedAttention 的间接寻址会导致约 5-10% 的 kernel 性能下降（相比连续内存），但换来的是 3-4 倍的内存利用率提升。在实际部署中，能装下更多并发请求 = 更高吞吐，远超 kernel 层面的开销。

Scheduler 在每个 iteration（一次模型前向传播）都要决定：哪些请求参与这次计算？

3.2 抢占策略：Recompute vs Swap

当 GPU 内存不足以容纳新请求时，Scheduler 必须抢占（preempt）正在运行的请求：

策略机制优点缺点适用场景
Recompute丢弃被抢请求的 KV Cache，需要时重新计算不占用 CPU 内存恢复延迟高（需重新 prefill）CPU 内存不足时
Swap将被抢请求的 KV Cache 换出到 CPU 内存恢复快（swap-in 即可）需要足够的 CPU 内存默认策略，CPU 内存充足时
抢占优先级：vLLM 默认使用 FCFS（先来先服务）策略 — 最晚到达的请求最先被抢占，因为它等待时间最短，被抢占的代价最低。

3.3 Scheduler._schedule() 完整伪代码

以下是 Scheduler 核心调度逻辑的完整伪代码，包含 can_allocate()、_preempt()、_swap() 的具体实现。基于 vLLM v1 源码结构。 
# ═══════════════════════════════════════════════════
# Scheduler._schedule() — 每个 iteration 调用一次
# ═══════════════════════════════════════════════════
def _schedule(self) -> SchedulerOutput:
    # ── Step 0: 初始化 ──
    budget = SchedulingBudget(
        max_num_tokens=self.max_num_batched_tokens,  # 如 4096
        max_num_seqs=self.max_num_seqs,               # 如 256
    )
    preempted_reqs = []
    swapped_in_reqs = []

    # ══════════════════════════════════════
    # Phase 1: 尝试将 Swapped 队列中的请求 swap-in 回 GPU
    # ══════════════════════════════════════
    while self.swapped:
        req = self.swapped[0]
        num_blocks_needed = len(req.block_table)

        # 检查 GPU 是否有足够空闲 block
        if not self.block_manager.can_swap_in(num_blocks_needed):
            # 尝试抢占 Running 中最低优先级的请求
            if not self._preempt_by_swap(req):
                break  # 无法腾出空间
            continue

        # 执行 swap-in: CPU → GPU
        self.block_manager.swap_in(req.block_table)
        self.swapped.pop(0)
        swapped_in_reqs.append(req)
        budget.add_num_seqs(1)
        budget.add_num_tokens(req.num_tokens)

    # ══════════════════════════════════════
    # Phase 2: 尝试调度 Waiting 队列中的新请求
    # ══════════════════════════════════════
    while self.waiting:
        req = self.waiting[0]

        # 检查预算
        if budget.num_seqs >= budget.max_num_seqs:
            break  # 达到最大并发数
        if budget.num_tokens + req.num_tokens > budget.max_num_tokens:
            break  # 达到单次 iteration 最大 token 数

        # 计算需要的 block 数量
        num_new_blocks = ceil(req.num_tokens / self.block_size)

        # ★ can_allocate() 检查
        if self.block_manager.can_allocate(num_new_blocks):
            # 分配物理 blocks
            blocks = self.block_manager.allocate(num_new_blocks)
            req.block_table = blocks
            self.waiting.pop(0)
            self.running.append(req)
            budget.add_num_seqs(1)
            budget.add_num_tokens(req.num_tokens)
        else:
            # 内存不足，尝试抢占
            if not self._preempt_by_swap_or_recompute():
                break  # 无法抢占，停止

    # ══════════════════════════════════════
    # Phase 3: Running 队列的 Decode 请求
    # ══════════════════════════════════════
    running_reqs = []
    for req in self.running:
        # Decode 每步生成 1 个 token
        if budget.num_tokens >= budget.max_num_tokens:
            break
        # 检查是否需要新 block（当前 block 已满）
        if req.num_tokens % self.block_size == 0:
            if self.block_manager.can_allocate(1):
                new_block = self.block_manager.allocate(1)
                req.block_table.append(new_block)
            else:
                # 内存不足 → 抢占
                self._preempt_by_swap_or_recompute(req)
                continue
        running_reqs.append(req)
        budget.add_num_tokens(1)

    return SchedulerOutput(
        scheduled_new_reqs=new_reqs,
        scheduled_running_reqs=running_reqs,
        preempted_reqs=preempted_reqs,
    )


# ═══════════════════════════════════════════════════
# _preempt() — 抢占机制
# ═══════════════════════════════════════════════════
def _preempt_by_swap(self, victim_req: Request) -> bool:
    """将被抢占请求的 KV Cache swap 到 CPU"""
    # 选择 victim: 最晚到达的请求（FCFS）
    victim = self.running[-1]  # 最后到达 = 最低优先级

    # 将 KV blocks 从 GPU 移到 CPU
    cpu_blocks = self.block_manager.swap_out(victim.block_table)
    victim.block_table = cpu_blocks
    victim.status = SWAPPED

    self.running.remove(victim)
    self.swapped.append(victim)
    return True  # 成功腾出空间

def _preempt_by_recompute(self, victim_req: Request) -> bool:
    """丢弃被抢占请求的 KV Cache，需要时重新计算"""
    victim = self.running[-1]

    # 直接释放物理 blocks（不 swap）
    self.block_manager.free(victim.block_table)
    victim.block_table = []
    victim.status = WAITING_FOR_RECOMPUTE

    self.running.remove(victim)
    self.waiting.insert(0, victim)  # 放回等待队列最前面
    return True


# ═══════════════════════════════════════════════════
# can_allocate() — 内存检查
# ═══════════════════════════════════════════════════
def can_allocate(self, num_blocks: int) -> bool:
    """检查是否有足够空闲物理 blocks"""
    free_blocks = self.num_free_gpu_blocks
    # 需要预留一些 blocks 给正在运行的请求的未来 decode
    reserved = self.num_running_reqs * self.blocks_per_reserve
    available = free_blocks - reserved
    return available >= num_blocks

4.2 KV Cache 容量计算

理解 KV Cache 能装多少 tokens，是调优的关键：

# KV Cache 容量计算公式
# 每个 token 的 KV Cache 大小:
bytes_per_token = 2 × num_layers × num_kv_heads × head_dim × dtype_bytes
#   × 2: K 和 V 各一份
#   dtype_bytes: FP16=2, FP8=1, INT4≈0.5

# 例: Qwen2.5-72B (80层, 8 KV heads, 128 head dim, FP16)
bytes_per_token = 2 × 80 × 8 × 128 × 2 = 327,680 bytes ≈ 320 KB

# 可用 KV Cache = GPU总显存 × gpu_memory_utilization - 模型权重 - 激活值
# H100 80GB, gpu_memory_utilization=0.9:
# 可用 ≈ 72GB - 144GB(权重需要TP分片) - 2GB(激活)
# 单卡 72B 需要 TP=2, 每卡 ~72GB 权重 → 可用 KV ≈ 80×0.9 - 72 - 2 ≈ -2GB
# → 72B 单卡跑不了！需要 TP=4 或量化

# 实际建议: 用 vLLM 的 --enforce-eager 启动，日志会打印 KV Cache 容量

4.2.1 完整公式推导

KV Cache 内存的完整公式： 
# ═══════════════════════════════════════════════════
# KV Cache 内存计算 — 完整公式
# ═══════════════════════════════════════════════════

# 单个 token 的 KV Cache 大小（字节）:
#
#   bytes_per_token = 2 × 2 × num_layers × num_kv_heads × head_dim × dtype_bytes
#                     ↑  ↑
#                     │  └── 第二个 2: 每个 head 需要 K 和 V 两份
#                     └── 第一个 2: 不是这个意思！见下面详细解释
#
# 等等，让我重新解释公式中的 2×2:

# 正确的分解:
#
# bytes_per_token = 2 (K和V) × num_layers × num_kv_heads × head_dim × dtype_bytes
#
# 为什么有时写 2×2？
# 那是因为 GQA (Grouped Query Attention) 的关系:
#   - MHA (Multi-Head Attention): num_kv_heads == num_heads
#   - GQA (Grouped Query Attention): num_kv_heads < num_heads
#   - MQA (Multi-Query Attention): num_kv_heads == 1
#
# 有些文章写 2×num_layers×num_heads×head_dim，那是 MHA 的特例
# vLLM 统一用 num_kv_heads，所以公式是:
#
# bytes_per_token = 2 × num_layers × num_kv_heads × head_dim × dtype_bytes
#
# 其中 2 = K cache + V cache 两份

# ═══════════════════════════════════════════════════
# 各模型的具体计算
# ═══════════════════════════════════════════════════

# ── Qwen2.5-7B (MHA) ──
# num_layers=28, num_kv_heads=4 (GQA), head_dim=128, FP16
bytes_per_token_7b = 2 × 28 × 4 × 128 × 2 = 57,344 bytes ≈ 56 KB
# 1000 并发 × 2048 tokens = 2,048,000 tokens
# KV 总量 ≈ 56KB × 2M = 112 GB → 需要多卡！
# 实际: H100 80GB, 权重 ~14GB, 可用 KV ≈ 58GB → 约 100 万 tokens
# → 约 500 并发 × 2048 tokens

# ── Qwen2.5-14B ──
# num_layers=40, num_kv_heads=8, head_dim=128, FP16
bytes_per_token_14b = 2 × 40 × 8 × 128 × 2 = 163,840 bytes ≈ 160 KB
# H100 80GB, 权重 ~28GB, 可用 KV ≈ 44GB
# → 约 27.5 万 tokens → 约 130 并发 × 2048 tokens

# ── Qwen2.5-72B ──
# num_layers=80, num_kv_heads=8, head_dim=128, FP16
bytes_per_token_72b = 2 × 80 × 8 × 128 × 2 = 327,680 bytes ≈ 320 KB
# H100 80GB × 4 (TP=4), 总权重 ~144GB (每卡 ~36GB)
# 每卡可用 KV ≈ 80×0.9 - 36 - 2 ≈ 34GB
# 4 卡总计 ≈ 136GB KV
# → 约 42.5 万 tokens → 约 200 并发 × 2048 tokens

# ── Llama-3.1-405B ──
# num_layers=128, num_kv_heads=8, head_dim=128, FP16
# H100 80GB × 8 (TP=8), 每卡权重 ~102GB/8 ≈ 12.75GB → 实际需要 TP=16 或量化
# FP8 量化: 权重 ~51GB/卡 → TP=8 每卡 ~51GB → 可用 KV ≈ 21GB/卡
# bytes_per_token = 2 × 128 × 8 × 128 × 2 = 524,288 bytes ≈ 512 KB (FP16)
# bytes_per_token = 2 × 128 × 8 × 128 × 1 = 262,144 bytes ≈ 256 KB (FP8)

4.2.2 与 gpu_memory_utilization 的关系

# vLLM 启动时的内存分配逻辑（伪代码）:
#
# total_gpu_memory = torch.cuda.get_device_properties().total_memory  # 如 80GB
# usable_memory = total_gpu_memory × gpu_memory_utilization            # 如 72GB
#
# model_weights_size = 参数量 × dtype_bytes
#   - FP16: 7B → 14GB, 14B → 28GB, 72B → 144GB (需 TP)
#   - INT8: 减半
#   - INT4/AWQ: 再减半
#
# cuda_overhead = ~0.5-1GB (CUDA context, cuBLAS workspace 等)
# activation_memory = ~1-4GB (与 batch_size 相关，通常可忽略)
#
# kv_cache_size = usable_memory - model_weights_size - cuda_overhead - activation_memory
# num_gpu_blocks = kv_cache_size ÷ (bytes_per_block)
# bytes_per_block = 2 × num_layers × num_kv_heads × head_dim × dtype_bytes × block_size
#
# 最终: 最大 tokens = num_gpu_blocks × block_size
#
# ⚠️ gpu_memory_utilization 不是 KV Cache 的比例，而是总显存的使用比例！
# 设太低 → KV Cache 少 → 并发低
# 设太高 → 可能 OOM（PyTorch 的碎片化内存、临时 tensor 等）
# 推荐: 0.85-0.92，视显存大小而定

Q1: PagedAttention 和传统 KV Cache 的核心区别是什么？
A: 传统方案为每个请求预分配连续的 KV Cache 内存（按最大长度），导致严重的内存浪费和碎片化。PagedAttention 借鉴操作系统的虚拟内存，将 KV Cache 分成固定大小的 Block，按需分配，不要求连续。通过 Block Table（类似页表）映射逻辑块到物理块，内存利用率从 20-40% 提升到 96%+。
Q2: vLLM 的 Scheduler 如何决定先处理哪个请求？
A: 默认 FCFS（先来先服务），优先 swap-in 被抢占的请求。当 GPU 内存不足时，抢占最晚到达的请求（Swap 或 Recompute）。可以通过 Chunked Prefill 让 Prefill 和 Decode 混合调度，避免长 Prefill 阻塞 Decode。
Q3: Copy-on-Write 在 vLLM 中如何工作？
A: 当多个请求共享相同前缀时（如 System Prompt），它们的 KV Cache blocks 通过引用计数共享同一份物理内存。只有当某个请求需要修改共享 block 时（生成新 token），才会复制一份。这在多轮对话和 Agent 场景中节省大量内存。
Q4: 如何估算一个模型需要多少 GPU 内存？
A: 三部分：(1) 模型权重：参数量 × dtype_bytes（FP16=2B，INT4≈0.5B）；(2) KV Cache：2 × layers × kv_heads × head_dim × dtype_bytes × 当前序列长度 × 并发数；(3) 激活值和框架开销：通常 1-4GB。例如 7B FP16 模型需要 ~14GB 权重 + N × KV 开销。单卡 24GB 可以跑 7B + 约 2000-4000 tokens 的 KV Cache。
Q5: vLLM 和 SGLang 的架构级区别？
A: vLLM 用 PagedAttention（基于 Block Table 的页式管理），适合通用场景。SGLang 用 RadixAttention（基于基数树的前缀管理），在多轮对话和 Agent 场景中前缀复用更智能。vLLM 社区更大、模型支持更广；SGLang 在结构化生成（JSON/正则约束）方面更强。
