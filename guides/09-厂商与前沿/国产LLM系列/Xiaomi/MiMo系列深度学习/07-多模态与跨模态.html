# DPO训练核心实现
import torch
from transformers import AutoModelForCausalLM

def compute_logprobs(model, input_ids, labels):
    """计算序列的对数概率"""
    logits = model(input_ids).logits
    log_probs = torch.nn.functional.log_softmax(logits, dim=-1)
    # 只计算response部分的log prob
    token_logps = log_probs.gather(2, labels.unsqueeze(-1)).squeeze(-1)
    return token_logps.sum(dim=-1)

def dpo_loss(policy, ref, chosen, rejected, beta=0.1):
    """DPO损失：无需奖励模型的偏好优化"""
    pi_chosen = compute_logprobs(policy, chosen['input_ids'], chosen['labels'])
    pi_rejected = compute_logprobs(policy, rejected['input_ids'], rejected['labels'])
    with torch.no_grad():
        ref_chosen = compute_logprobs(ref, chosen['input_ids'], chosen['labels'])
        ref_rejected = compute_logprobs(ref, rejected['input_ids'], rejected['labels'])

    # 核心公式：隐式奖励差的sigmoid
    log_ratio_chosen = pi_chosen - ref_chosen
    log_ratio_rejected = pi_rejected - ref_rejected
    loss = -torch.nn.functional.logsigmoid(beta * (log_ratio_chosen - log_ratio_rejected))
    return loss.mean()

视觉编码器的数学原理

MiMo-V2-Omni采用SigLIP作为视觉编码器，其数学原理基于对比学习的损失函数。给定图像-文本对(x_i, t_i)，SigLIP使用sigmoid损失而非softmax损失：L = -log(σ(y_ij · (z_i^T · t_j / τ)))，其中z_i是图像嵌入，t_j是文本嵌入，τ是温度参数。这一公式推导体现了二元交叉熵损失在多模态对齐中的优势。论文中通过消融实验证明，SigLIP在MiMo架构中相比CLIP提升了3.2%的视觉问答准确率。

    实际案例: MiMo-V2-Omni的视频处理在骁龙8 Gen4端侧芯片上无法实时运行完整模型，小米的解决方案是：端侧仅运行视频关键帧提取和基础物体检测(2B量化版)，完整的视频理解(时序分析、音频联动)交由边缘端或云端处理。诊断性能瓶颈的关键指标是"每秒处理的模态token数"——如果视频token处理速率低于实时速率(25token/s)，说明编码器是瓶颈。

  3. 核心原理: MiMo-V2-Omni的替代方案包括：(1)独立模型组合——为每种模态训练独立的专家模型，然后组合使用，优势是每个模型可独立优化到最优，劣势是跨模态交互需要额外的桥接层且组合推理延迟高；(2)两阶段VLM(视觉语言模型)——先用独立视觉/音频编码器处理非文本信号，再将特征序列输入LLM，是当前主流方案(如GPT-4V)，优势是工程成熟、可利用预训练编码器，劣势是跨模态融合浅层(仅在输入层注入)；(3)级联模型——大模型负责推理、小模型负责特定模态编码，优势是灵活替换组件，劣势是流水线延迟累积；(4)后续融合架构——保持预训练LLM不变，仅训练轻量级的跨模态适配器，优势是低成本复用LLM，劣势是融合能力受限。
    关键要点: ①独立模型组合适合模态间交互需求低的场景；②两阶段VLM(如GPT-4V)适合图文为主、音频视频需求少的场景；③级联模型适合需要频繁更新模态编码器的快速迭代场景；④MiMo-V2-Omni的全模态深度融合适合强跨模态需求(如视频理解+音频分析)；⑤选择依据：模态耦合度要求高→全模态原生；要求低→独立组合或两阶段
    实际案例: 在构建视频字幕生成系统时，若仅需理解视频内容生成文本描述，两阶段VLM(视觉编码器+LLM)即可满足且成本低；但若需要"识别视频中特定人物的声音+理解其与画面中人物唇形的一致性"，则需要MiMo-V2-Omni这样的全模态模型——这是独立模型组合难以胜任的。
    关联知识: 从趋势看，全模态原生模型(类似MiMo-V2-Omni)正成为行业方向，但工程上两阶段VLM因实现简单仍是主流选择。Xiaomi选择全模态路线的战略考量是其IoT生态(智能家居、汽车)天然需要多模态深度融合。

# MiMo-V2-Omni 性能基准测试
import torch
import time

def benchmark(model, input_data, warmup=10, iters=100):
    for _ in range(warmup):
        model(input_data)
    torch.cuda.synchronize()
    start = time.perf_counter()
    for _ in range(iters):
        model(input_data)
    torch.cuda.synchronize()
    elapsed = (time.perf_counter() - start) / iters
    print(f"Latency: {elapsed*1000:.2f}ms")
    return elapsed

MiMo-V2-OmniXiaomi技术全景总结

MiMo-V2-Omni作为小米的全模态模型，支持视觉、音频、视频和文本四种模态的统一理解与生成，其架构采用"模态特定编码器+统一Token空间+跨模态融合层"的三层设计，实现了跨模态的深度融合而非简单的特征拼接。视觉编码器基于ViT-Large（约3亿参数，patch_size=14），支持动态分辨率输入（336×336到1344×1344），将图像编码为256到9216个视觉token；音频编码器采用类Whisper架构（约6千万参数），将16kHz音频波形通过2层1D卷积下采样后送入Transformer Encoder，输出每秒约25个音频token；视频编码器基于时序ViT（TimeSformer变体），以2fps采样后编码，每帧产生256个视觉token加上时序位置编码和运动特征token。跨模态融合层使用门控交叉注意力机制，Query来自语言模型隐状态，Key/Value来自各模态编码器输出，门控参数α（可学习，初始化为0.1）控制模态信息注入强度，避免训练初期模态信息淹没文本信号。训练分三阶段渐进式展开：第一阶段仅训练连接层（约2B tokens图文对，学习率1e-3），第二阶段解冻视觉编码器（约500M tokens多模态数据，学习率2e-5），第三阶段全模型微调（约100M tokens高质量多模态指令数据，学习率5e-6）。在小米生态部署上采用端-边-云三层协同架构：云端部署全模态大模型处理复杂推理（延迟约800ms），边缘端部署7B蒸馏版处理实时任务（延迟约50ms），端侧部署2B量化版（INT4，显存约1.2GB）处理语音唤醒和基础视觉任务（延迟小于20ms）。端云协同通过"中间特征传输"创新——端侧完成初步推理后将中间特征（而非原始数据）上传云端，减少传输数据量约60%同时保护用户隐私。PinchBench全模态基准中MiMo-V2-Omni在音频理解赛道排名第一（准确率89.3%），视频理解赛道排名第二（准确率86.7%），充分验证了全模态统一架构的有效性。
