# 使用 LoRA 微调 Baichuan2-7B-Chat 的基本框架
from transformers import AutoTokenizer, AutoModelForCausalLM
from peft import LoraConfig, get_peft_model, TaskType
from trl import SFTTrainer
from datasets import load_dataset

# 加载模型
model_name = "baichuan-inc/Baichuan2-7B-Chat"
tokenizer = AutoTokenizer.from_pretrained(model_name, trust_remote_code=True)
model = AutoModelForCausalLM.from_pretrained(
    model_name, trust_remote_code=True,
    device_map="auto", torch_dtype="auto"
)

# LoRA 配置
lora_config = LoraConfig(
    task_type=TaskType.CAUSAL_LM,
    r=8,                    # LoRA 秩
    lora_alpha=32,          # 缩放系数
    lora_dropout=0.05,
    target_modules=["W_pack", "o_proj", "gate_proj", "up_proj", "down_proj"],
)

model = get_peft_model(model, lora_config)
model.print_trainable_parameters()
# 输出: trainable params: ~0.1% || all params: 7B

# 加载训练数据
dataset = load_dataset("your_dataset", split="train")

# 开始微调
trainer = SFTTrainer(
    model=model,
    train_dataset=dataset,
    tokenizer=tokenizer,
    max_seq_length=2048,
    args=TrainingArguments(
        output_dir="./baichuan2-7b-lora",
        num_train_epochs=3,
        per_device_train_batch_size=4,
        gradient_accumulation_steps=4,
        learning_rate=2e-4,
        warmup_steps=100,
        logging_steps=10,
        save_steps=500,
        bf16=True,
    ),
)
trainer.train()

4.3 学习注意事项

⚠️ 学习过程中的注意事项
  • 显存管理：7B 模型 FP16 约需 14GB 显存，13B 约需 26GB，优先考虑量化方案
  • 对话模板：Baichuan2-Chat 有专用对话模板，错误使用会导致输出质量下降
  • 数据质量：微调时数据质量远比数据数量重要，参考百川的数据筛选方法论
  • 许可协议：Baichuan2 社区许可证限制商用场景，请仔细阅读条款
  • 评测客观性：对比模型时保持相同的评测条件（few-shot、prompt 格式等）
