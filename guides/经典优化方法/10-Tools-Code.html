1. 求解器对比

求解器对比

通俗类比：选择求解器就像为任务挑选合适的工具——用扳手拧螺丝效率低，用螺丝刀敲钉子也不对。了解工具特性，才能事半功倍。
常见误区：误区：盲目追求‘最快’或‘最通用’的求解器。实际上，求解器性能高度依赖问题结构，选错可能导致不收敛或计算时间激增。
深入探讨：以线性规划为例，单纯形法平均复杂度约为O(m^2n)，而内点法可达O(n^3.5)。对于稀疏矩阵，内点法常更高效；但对于稠密小规模问题，单纯形法可能更稳定。

示例1：LP — 生产计划 (PuLP)

from pulp import *

prob = LpProblem("Production", LpMaximize)
x1 = LpVariable("Product_A", lowBound=0)  # 产品A产量
x2 = LpVariable("Product_B", lowBound=0)  # 产品B产量

prob += 5*x1 + 4*x2, "Profit"             # 利润最大化
prob += 6*x1 + 4*x2 <= 24, "Machine"      # 机器工时
prob += x1 + 2*x2 <= 6, "Labor"           # 人工

prob.solve()
print(f"A={x1.value():.1f}, B={x2.value():.1f}, 利润={value(prob.objective):.1f}")
# 灵敏度分析
for name, c in prob.constraints.items():
    print(f"  {name}: shadow_price={c.pi:.2f}, slack={c.slack:.2f}")

示例2：MIP — 设施选址 (PuLP)

from pulp import *

# 3个候选仓库, 4个客户
fixed = [100, 120, 90]              # 建设固定成本
trans = [[8,6,10,9],[9,12,7,8],[14,9,16,5]]  # 运输成本
demand = [80, 90, 70, 60]

prob = LpProblem("FacilityLocation", LpMinimize)
y = [LpVariable(f"Open_{j}", cat="Binary") for j in range(3)]
x = [[LpVariable(f"Assign_{i}_{j}", lowBound=0, upBound=1) for j in range(3)] for i in range(4)]

prob += (sum(fixed[j]*y[j] for j in range(3))
       + sum(trans[j][i]*x[i][j] for i in range(4) for j in range(3)))

for i in range(4):
    prob += sum(x[i][j] for j in range(3)) == 1  # 每客户必须被服务
for i in range(4):
    for j in range(3):
        prob += x[i][j] <= y[j]  # 只有开了仓库才能服务

prob.solve()
print(f"开设仓库: {[j for j in range(3) if y[j].value() > 0.5]}")
print(f"总成本: {value(prob.objective):.0f}")

示例3：NLP — Rosenbrock函数 (SciPy)

from scipy.optimize import minimize

rosenbrock = lambda x: (1-x[0])**2 + 100*(x[1]-x[0]**2)**2

for method in ['L-BFGS-B', 'Nelder-Mead', 'Powell']:
    res = minimize(rosenbrock, [0,0], method=method)
    print(f"{method:15s}: x={res.x}, f={res.fun:.6f}, iters={res.nit}")

示例4：凸优化 — SVM (CVXPY)

import cvxpy as cp
import numpy as np

# 线性SVM: min ||w||² + C*Σξᵢ
n, d = 100, 5
X = np.random.randn(n, d)
y = np.sign(X @ np.array([1, -1, 0, 0, 0]) + 0.1*np.random.randn(n))

w = cp.Variable(d)
b = cp.Variable()
xi = cp.Variable(n)

prob = cp.Problem(cp.Minimize(cp.sum_squares(w) + 10*cp.sum(xi)),
                  [y[i]*(X[i]@w + b) >= 1 - xi[i] for i in range(n)] + [xi >= 0])
prob.solve()
print(f"w={w.value}, b={b.value:.3f}")

import networkx as nx

# 最短路径
G = nx.DiGraph()
G.add_weighted_edges_from([('A','B',1),('A','C',4),('B','C',2),('B','D',6),('C','D',3)])
dist, path = nx.single_source_dijkstra(G, 'A', 'D')
print(f"最短路: {'→'.join(path)}, 距离: {dist}")

# 最大流
F = nx.DiGraph()
F.add_edge('S','A',capacity=5); F.add_edge('S','B',capacity=4)

F.add_edge('A','T',capacity=3); F.add_edge('C','T',capacity=6)
flow_value, flow_dict = nx.maximum_flow(F, 'S', 'T')
print(f"最大流: {flow_value}")

示例6：启发式 — 模拟退火TSP

import random, math

def tsp_sa(dist, n_iter=100000, T0=1000, cooling=0.9995):
    n = len(dist)
    tour = list(range(n))
    random.shuffle(tour)
    best = tour[:]
    best_cost = sum(dist[tour[i]][tour[(i+1)%n]] for i in range(n))
    T = T0
    for _ in range(n_iter):
        i, j = sorted(random.sample(range(n), 2))
        new = tour[:i] + tour[i:j+1][::-1] + tour[j+1:]  # 2-opt
        new_cost = sum(dist[new[k]][new[(k+1)%n]] for k in range(n))
        delta = new_cost - best_cost
        if delta < 0 or random.random() < math.exp(-delta/T):
            tour = new
            if new_cost < best_cost:
                best, best_cost = tour[:], new_cost
        T *= cooling
    return best, best_cost

# 随机10城市TSP
import numpy as np
cities = np.random.rand(10, 2)
d = [[np.linalg.norm(cities[i]-cities[j]) for j in range(10)] for i in range(10)]
tour, cost = tsp_sa(d)
print(f"最优路线: {tour}, 总距离: {cost:.2f}")

示例7：装箱 — FFD算法

def first_fit_decreasing(items, capacity):
    items = sorted(items, reverse=True)  # 从大到小
    bins = []
    for item in items:
        placed = False
        for b in bins:
            if sum(b) + item <= capacity:
                b.append(item)
                placed = True
                break
        if not placed:
            bins.append([item])
    return bins

items = [4, 8, 1, 4, 2, 1, 7, 3, 5, 6, 2, 9]
bins = first_fit_decreasing(items, capacity=10)
for i, b in enumerate(bins):
    print(f"箱子{i+1}: {b} (总大小: {sum(b)}/{10})")
print(f"总共 {len(bins)} 个箱子")
