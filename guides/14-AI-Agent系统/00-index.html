🧪 练习 1：RAG Pipeline 端到端实现

请用 Python 实现一个完整的 RAG 流水线：加载文档 → 分块 → Embedding → 存入向量数据库 → 检索 + 生成。要求使用 FAISS 作为向量库，HuggingFace Embedding 模型，以及一个本地 LLM (可用 HuggingFace pipeline 模拟)。

💡 参考答案与详解

解题思路: 将 RAG 拆解为离线索引和在线检索两阶段。离线阶段负责文档加载、分块、向量化入库；在线阶段负责查询编码、相似度检索、上下文注入和生成。核心是分块策略的选择和检索质量的评估。

import numpy as np
from typing import List, Dict
from sentence_transformers import SentenceTransformer
import faiss

# ============ 离线：构建向量索引 ============
class RAGIndexer:
    def __init__(self, model_name: str = "BAAI/bge-small-zh-v1.5"):
        self.encoder = SentenceTransformer(model_name)
        self.dimension = self.encoder.get_sentence_embedding_dimension()
        self.index = faiss.IndexFlatIP(self.dimension)  # 内积 = 余弦相似度（归一化后）
        self.documents: List[str] = []
        self.metadata: List[Dict] = []

    def chunk_text(self, text: str, chunk_size: int = 200, overlap: int = 40) -> List[str]:
        """滑动窗口分块：保持语义连贯的同时控制块大小"""
        chunks = []
        start = 0
        while start < len(text):
            end = min(start + chunk_size, len(text))
            chunks.append(text[start:end])
            start += chunk_size - overlap
        return chunks

    def build_index(self, documents: List[Dict[str, str]]):
        """构建 FAISS 索引：文档 → 分块 → 编码 → 入库"""
        all_chunks = []
        all_meta = []

        for doc in documents:
            chunks = self.chunk_text(doc["content"])
            for chunk in chunks:
                all_chunks.append(chunk)
                all_meta.append({"source": doc["title"], "chunk": chunk[:50] + "..."})

        # 批量编码（GPU 加速，比逐条编码快 10x+）
        embeddings = self.encoder.encode(
            all_chunks, normalize_embeddings=True,  # L2 归一化后内积=余弦相似度
            batch_size=64, show_progress_bar=True
        )

        self.index.add(np.array(embeddings).astype('float32'))
        self.documents = all_chunks
        self.metadata = all_meta
        print(f"索引构建完成：{len(all_chunks)} 个 chunk，维度 {self.dimension}")

# ============ 在线：检索 + 生成 ============
class RAGRetriever:
    def __init__(self, indexer: RAGIndexer, top_k: int = 3):
        self.indexer = indexer
        self.top_k = top_k

    def retrieve(self, query: str) -> List[Dict]:
        """语义检索：编码 query → ANN 搜索 → 返回 top-k 相关文档"""
        query_vec = self.indexer.encoder.encode(
            [query], normalize_embeddings=True
        ).astype('float32')

        scores, indices = self.indexer.index.search(query_vec, self.top_k)

        results = []
        for score, idx in zip(scores[0], indices[0]):
            results.append({
                "content": self.indexer.documents[idx],
                "score": float(score),
                "meta": self.indexer.metadata[idx]
            })
        return results

    def generate(self, query: str, use_llm: bool = True) -> str:
        """检索增强生成：拼接 prompt → 调用 LLM"""
        docs = self.retrieve(query)
        context = "\n\n".join([f"[来源: {d['meta']['source']}] {d['content']}" for d in docs])

        prompt = f"""基于以下参考材料回答问题。如果材料中没有相关信息，请明确说明。

参考材料:
{context}

问题: {query}

回答:"""

        if use_llm:
            # 实际场景：调用 vLLM / Ollama / OpenAI API
            # return call_llm(prompt)
            return f"[LLM 将基于以下上下文生成回答]\n上下文长度: {len(context)} 字符\nTop-1 相关性: {docs[0]['score']:.3f}"
        return prompt  # 调试模式直接返回 prompt

# ============ 运行示例 ============
if __name__ == "__main__":
    # 准备知识库
    docs = [
        {"title": "Agent 定义", "content": "Agent 是感知环境并通过执行器作用于环境的实体。AI Agent 使用 LLM 作为推理引擎，具备自主规划、记忆、工具使用能力。"},
        {"title": "RAG 架构", "content": "RAG 检索增强生成由三个部分组成：检索器从外部知识库获取相关信息，编码器将文本转为向量，生成器结合检索结果和用户输入生成最终回答。"},
        {"title": "向量数据库", "content": "FAISS 是 Meta 开发的向量相似度搜索库，支持多种索引类型。IndexFlatIP 使用内积度量，适合中小规模精确搜索。HNSW 和 IVF 适合大规模近似搜索。"},
        {"title": "分块策略", "content": "文档分块是 RAG 的关键步骤。过小的块会丢失上下文，过大的块引入噪声。常用策略包括：固定大小分块、语义分块、递归分块和滑动窗口重叠分块。"},
    ]

    # 构建索引
    indexer = RAGIndexer()
    indexer.build_index(docs)

    # 检索生成
    retriever = RAGRetriever(indexer, top_k=2)
    query = "RAG 系统如何构建向量索引？"
    answer = retriever.generate(query, use_llm=False)
    print(f"\n查询: {query}\n{answer}")

关键知识点: (1) 分块策略直接影响检索质量——重叠窗口保持语义连贯；(2) FAISS IndexFlatIP + L2 归一化向量 = 余弦相似度计算；(3) 检索后重排序(Rerank)是生产必备步骤；(4) 中文场景推荐 bge-large-zh 或 m3e-base 作为 Embedding 模型；(5) 评估 RAG 质量的指标包括 Recall@k、忠实度(faithfulness)、相关性(relevance)。
